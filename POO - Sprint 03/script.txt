/*Criando tabela ENDEREÇO*/
CREATE TABLE ENDERECO(

    cidade NUMBER(10) NOT NULL
);

ALTER TABLE ENDERECO ADD CONSTRAINT pk_endereco PRIMARY KEY(id);
ALTER TABLE ENDERECO ADD CONSTRAINT fk_endereco FOREIGN KEY(cidade)
